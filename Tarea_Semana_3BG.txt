  descargar_hf <- function(cfg, max_intentos = 3) {
    
    url_api <- modify_url(
      cfg$url_base,
      query = list(
        dataset = cfg$dataset_id,
        config  = cfg$config_dataset,
        split   = cfg$split_dataset,
        offset  = cfg$offset,
        length  = cfg$n_registros
      )
    )
    log_info("URL:", url_api)
    
    hf_token <- Sys.getenv("HF_TOKEN")
    hdrs <- c("User-Agent" = "RStudio-Auditoria-IA-v2")
    if (nzchar(hf_token)) {
      hdrs <- c(hdrs, "Authorization" = paste("Bearer", hf_token))
      log_info("Token HF detectado.")
    } else {
      log_warn("Sin HF_TOKEN. Si el dataset exige aceptar terminos, fallara.")
    }
    
    for (intento in seq_len(max_intentos)) {
      log_info(glue("Descarga, intento {intento}/{max_intentos}..."))
      
      resp <- tryCatch(
        GET(url_api, do.call(add_headers, as.list(hdrs)), timeout(60)),
        error = function(e) { log_error("Red:", conditionMessage(e)); NULL }
      )
      
      if (!is.null(resp) && status_code(resp) == 200) {
        cont <- fromJSON(content(resp, "text", encoding = "UTF-8"),
                         simplifyDataFrame = TRUE)
        if (!is.null(cont$rows) && length(cont$rows) > 0) {
          df <- cont$rows$row
          log_info("OK. Filas descargadas:", nrow(df))
          return(df)
        }
      } else if (!is.null(resp)) {
        log_warn("HTTP", status_code(resp))
      }
      Sys.sleep(2)
    }
    
    stop("No se pudo descargar tras ", max_intentos, " intentos.")
  }

  # Despues de fromJSON(simplifyDataFrame=TRUE), suele venir como list-column
  # donde cada elemento es un data.frame con columnas role y content.
  # Manejamos ambos casos (data.frame y lista) por seguridad.
  
  extraer_mensaje <- function(conv, rol = "user", indice = 1) {
    if (is.null(conv) || length(conv) == 0) return(NA_character_)
    
    if (is.data.frame(conv)) {
      if (!all(c("role", "content") %in% names(conv))) return(NA_character_)
      msgs <- conv$content[conv$role == rol]
      if (length(msgs) >= indice) return(as.character(msgs[[indice]]))
      return(NA_character_)
    }
    
    if (is.list(conv)) {
      roles    <- map_chr(conv, ~ as.character(.x$role    %||% NA))
      contents <- map_chr(conv, ~ as.character(.x$content %||% NA))
      msgs     <- contents[roles == rol]
      if (length(msgs) >= indice) return(msgs[[indice]])
      return(NA_character_)
    }
    NA_character_
  }
  
  contar_turnos <- function(conv) {
    if (is.data.frame(conv)) return(nrow(conv))
    if (is.list(conv))       return(length(conv))
    0L
  }
  
  # Moderation de OpenAI: devolver TRUE si ALGUN turno fue flagged,
  # y la lista de categorias detectadas (separadas por ;).
  parse_moderation <- function(mod) {
    tryCatch({
      if (is.null(mod) || length(mod) == 0) {
        return(list(flagged = FALSE, cats = ""))
      }
      aplanado <- unlist(mod)
      nombres  <- names(aplanado) %||% ""
      
      # flagged en cualquier turno
      idx_flag <- str_ends(nombres, "flagged") | nombres == "flagged"
      flagged  <- any(as.logical(aplanado[idx_flag]) %in% TRUE)
      
      # categorias marcadas como TRUE
      idx_cat <- str_detect(nombres, "categories\\.")
      vals    <- suppressWarnings(as.logical(aplanado[idx_cat]))
      cats    <- nombres[idx_cat][vals %in% TRUE]
      cats    <- str_remove(cats, ".*categories\\.")
      
      list(flagged = flagged, cats = paste(unique(cats), collapse = ";"))
    }, error = function(e) list(flagged = FALSE, cats = ""))
  }

  # ---------------------------------------------------------
  # 8. CLASIFICACION DE RIESGO PONDERADA
  # ---------------------------------------------------------
  # Cada categoria tiene un regex con \b word boundaries y un peso.
  # El score total es la suma de pesos de categorias detectadas.
  # Esto evita falsos positivos tipo "hackathon" -> "hack".
  
  PATRONES_RIESGO <- list(
    credenciales = list(
      regex = "\\b(api[ _-]?key|password|contrase\u00f1a|token|secret|credential)\\b",
      peso  = 3
    ),
    cyber_attack = list(
      regex = "\\b(malware|virus|phishing|ransomware|exploit|vulnerability|backdoor)\\b",
      peso  = 3
    ),
    identidad = list(
      regex = "\\b(deepfake|fake\\s+identity|identidad\\s+falsa|impersonate)\\b",
      peso  = 3
    ),
    datos_personales = list(
      regex = "\\b(ssn|social\\s+security|datos\\s+personales|personal\\s+data|private\\s+information|informaci\u00f3n\\s+privada)\\b",
      peso  = 2
    ),
    evasion_prompt = list(
      regex = "\\b(bypass|jailbreak|ignore\\s+(previous|prior|above)\\s+(instructions|rules)|prompt\\s+injection)\\b",
      peso  = 3
    ),
    manipulacion = list(
      regex = "\\b(fake\\s+news|noticia\\s+falsa|propaganda)\\b",
      peso  = 2
    ),
    academico = list(
      regex = "\\b(tesis|thesis|ensayo|essay|tarea|homework|examen|exam)\\b",
      peso  = 1
    ),
    codigo = list(
      regex = "\\b(code|c\u00f3digo|program|script|function|algorithm)\\b",
      peso  = 1
    )
  )
  
  evaluar_riesgo <- function(prompt) {
    if (is.na(prompt) || nchar(prompt) == 0) {
      return(list(score = 0, cats = ""))
    }
    texto <- tolower(prompt)
    score <- 0
    detect <- character(0)
    for (nm in names(PATRONES_RIESGO)) {
      p <- PATRONES_RIESGO[[nm]]
      if (str_detect(texto, p$regex)) {
        score <- score + p$peso
        detect <- c(detect, nm)
      }
    }
    list(score = score, cats = paste(detect, collapse = ";"))
  }
  
  eval_list <- map(logs_ia$prompt, evaluar_riesgo)
  logs_ia$score_riesgo      <- map_dbl(eval_list, "score")
  logs_ia$categorias_riesgo <- map_chr(eval_list, "cats")

    fn <- sum(!logs_ia$nuestro_flag &  logs_ia$mod_flagged)
    tn <- sum(!logs_ia$nuestro_flag & !logs_ia$mod_flagged)
    
    precision <- if ((tp + fp) > 0) tp / (tp + fp) else NA_real_
    recall    <- if ((tp + fn) > 0) tp / (tp + fn) else NA_real_
    f1        <- if (!is.na(precision) && !is.na(recall) && (precision + recall) > 0)

      "Precision: {round(precision, 3)} | Recall: {round(recall, 3)} | ",
      "F1: {round(f1, 3)} | Accuracy: {round(accuracy, 3)}"
    ))
  } else {

  escribir_csv <- function(df, ruta) {
    df_safe <- df %>%
      select(-where(is.list)) %>%
      mutate(across(where(is.character), ~ str_replace_all(.x, "[\r\n]+", " ")))
    write_csv(df_safe, ruta)
  }

    cat("2. DISTRIBUCION DE RIESGO\n")
    print(knitr::kable(
      df %>% count(riesgo) %>% mutate(pct = round(n / sum(n) * 100, 1)),
      format = "simple"))
    cat("\n")
    
    cat("3. CATEGORIAS DETECTADAS MAS FRECUENTES\n")
    cats <- df %>% filter(nchar(categorias_riesgo) > 0) %>%
      separate_rows(categorias_riesgo, sep = ";") %>%
      count(categorias_riesgo, sort = TRUE) %>%
      slice_head(n = 10)
    if (nrow(cats) > 0) print(knitr::kable(cats, format = "simple"))
    else cat("   (sin detecciones)\n")
    cat("\n")
    
    cat("4. VALIDACION VS OPENAI MODERATION\n")
    if (any(df$mod_flagged)) {
      tp <- sum( df$nuestro_flag &  df$mod_flagged)
      fp <- sum( df$nuestro_flag & !df$mod_flagged)
      fn <- sum(!df$nuestro_flag &  df$mod_flagged)
      cat(sprintf("   Flagged OpenAI       : %d\n", sum(df$mod_flagged)))
      cat(sprintf("   Flagged clasificador : %d\n", sum(df$nuestro_flag)))

      cat(sprintf("   Falsos positivos (FP): %d\n", fp))

    } else {
      cat("   La muestra no incluye prompts flagged por OpenAI.\n")
    }
    cat("\n")
    
    cat("5. TOP 10 MODELOS POR VOLUMEN\n")
    print(knitr::kable(df %>% count(modelo, sort = TRUE) %>% slice_head(n = 10),
                       format = "simple"))
    cat("\n")
    
    cat("6. RECOMENDACIONES\n")
    cat("   - Revisar manualmente los prompts marcados como ALTO.\n")
    cat("   - Ajustar pesos y patrones con base en falsos positivos del set.\n")
    cat("   - Ampliar reglas para idiomas distintos al ingles.\n")
    cat("   - Implementar logging continuo en produccion (esto es batch).\n")
    cat("   - Auditar tambien las RESPUESTAS del modelo, no solo los prompts.\n")
    cat("\n", strrep("=", 64), "\n", sep = "")
  }
  
  generar_reporte(logs_ia, CONFIG,
                  file.path(CONFIG$proyecto_dir, "reportes", "reporte_ejecutivo.txt"))
  
  
  # ---------------------------------------------------------
  # 14. DASHBOARD EN CONSOLA
  # ---------------------------------------------------------
  cat("\n", strrep("=", 60), "\n", sep = "")
  cat(" DASHBOARD AUDITORIA IA GENERATIVA\n")
  cat(strrep("=", 60), "\n", sep = "")
  cat(sprintf(" Interacciones       : %d\n", nrow(logs_ia)))
  cat(sprintf(" Riesgo ALTO         : %d (%.1f%%)\n",
              sum(logs_ia$riesgo == "ALTO"),  mean(logs_ia$riesgo == "ALTO")  * 100))
  cat(sprintf(" Riesgo MEDIO        : %d (%.1f%%)\n",
              sum(logs_ia$riesgo == "MEDIO"), mean(logs_ia$riesgo == "MEDIO") * 100))
  cat(sprintf(" Riesgo BAJO         : %d (%.1f%%)\n",
              sum(logs_ia$riesgo == "BAJO"),  mean(logs_ia$riesgo == "BAJO")  * 100))
  cat(sprintf(" Bloqueados          : %d\n", sum(logs_ia$estado_prompt == "BLOQUEADO")))
  cat(sprintf(" Flagged por OpenAI  : %d\n", sum(logs_ia$mod_flagged)))
  cat(sprintf(" Modelos distintos   : %d\n", length(unique(logs_ia$modelo))))
  cat(sprintf(" Idiomas detectados  : %d\n", length(unique(logs_ia$idioma))))
  cat(strrep("=", 60), "\n", sep = "")
  
  log_info("Taller v2 finalizado. Carpeta:", CONFIG$proyecto_dir)
  stop("No se encontraron datos en la respuesta de Hugging Face.")
}

extraer_prompt <- function(conversation){
  
  if(is.null(conversation) || is.na(conversation)){
    return(NA)
  }
  
  texto <- as.character(conversation)
  
  # Buscar el primer mensaje del usuario
  if(str_detect(texto, "user")){
    return(str_sub(texto, 1, 1000))
  } else {
    return(str_sub(texto, 1, 1000))
  }
}

extraer_respuesta <- function(conversation){
  
  if(is.null(conversation) || is.na(conversation)){
    return(NA)
  }
  
  texto <- as.character(conversation)
  
  # Tomamos una parte posterior de la conversacion como respuesta aproximada
  return(str_sub(texto, 1001, 2000))
}
